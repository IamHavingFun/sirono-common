# Getting Contact record types
select name, id from recordtype where sobjecttype = 'Contact' # This should be queried against the target object, not the source like the rest of the queries here.

# Location
select name, sPRS__sPRS__description__c, sPRS__sPRS__external_description__c, sPRS__sPRS__external_id__c, sPRS__sPRS__origin_code__c, sPRS__sPRS__sirono_id__c, sPRS__sPRS__unmapped__c from sPRS__sPRS__location__c

# Payor
select name, sPRS__sPRS__description__c, sPRS__sPRS__external_description__c, sPRS__origin_code__c, sPRS__sirono_id__c, sPRS__unique_id__c, sPRS__unmapped__c, sPRS__external_id__c from sPRS__payor__c

# Provider
select name, sPRS__description__c, sPRS__external_description__c, sPRS__external_id__c, sPRS__origin_code__c, sPRS__sirono_id__c, sPRS__unmapped__c from sPRS__provider__c

# Service Rollup
select name, sPRS__description__c, sPRS__detail_description__c, sPRS__external_description__c, sPRS__external_id__c, sPRS__sirono_id__c, sPRS__unmapped__c from sPRS__service_rollup__c

# Account
select name, type, billingstreet, billingcity, billingstate, billingpostalcode, billingcountry, billinglatitude, billinglongitude, shippingstreet, shippingcity, shippingstate, shippingpostalcode, shippingcountry, shippinglatitude, shippinglongitude, phone, fax, accountnumber, website, photourl, sic, industry, annualrevenue, numberofemployees, ownership, tickersymbol, description, rating, site, jigsaw, jigsawcompanyid, accountsource, sicdesc, sPRS__external_account_name__c from account

# Guarantor contacts
select account.sPRS__external_account_name__c, lastname, firstname, middlename, suffix, recordtype.name, otherstreet, othercity, otherstate, otherpostalcode, mailingstreet, mailingcity, mailingstate, mailingpostalcode, mobilephone, homephone, otherphone, email, birthdate, sPRS__deleted__c, sPRS__external_guarantor_id__c, sPRS__external_id__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__invoice_offset_date__c, sPRS__medical_record_number__c, sPRS__origin_code__c, sPRS__Guarantor_Id__c, sPRS__sex__c, sPRS__social_security_number__c from contact where recordtype.name = 'Guarantor' and sPRS__Guarantor_Id__c >= 200000

# Patient contacts
select account.sPRS__external_account_name__c, lastname, firstname, middlename, suffix, recordtype.name, otherstreet, othercity, otherstate, otherpostalcode, mailingstreet, mailingcity, mailingstate, mailingpostalcode, mobilephone, homephone, otherphone, email, birthdate, sPRS__deleted__c, sPRS__external_guarantor_id__c, sPRS__external_id__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__invoice_offset_date__c, sPRS__medical_record_number__c, sPRS__origin_code__c, sPRS__patient_id__c, sPRS__Guarantor_Id__c, sPRS__sex__c, sPRS__social_security_number__c from contact where recordtype.name = 'Patient' and id in (select sPRS__patient__c from sPRS__charge_group__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000)

# External Guarantor contacts
select account.sPRS__external_account_name__c, lastname, firstname, middlename, suffix, recordtype.name, otherstreet, othercity, otherstate, otherpostalcode, mailingstreet, mailingcity, mailingstate, mailingpostalcode, mobilephone, homephone, otherphone, email, birthdate, sPRS__deleted__c, sPRS__external_guarantor_id__c, sPRS__external_id__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__invoice_offset_date__c, sPRS__medical_record_number__c, sPRS__origin_code__c, sPRS__patient_id__c, sPRS__Guarantor_Id__c, sPRS__sex__c, sPRS__social_security_number__c from contact where recordtype.name = 'External Guarantor' and sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Payment Method
select name, sPRS__expiration_date__c, sPRS__card_holder_name__c, sPRS__payment_method_token__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__card_type__c, sPRS__sirono_id__c, last_4__c from sPRS__payment_method__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Invoice
select name, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__due_date__c, sPRS__date__c, sPRS__patient__r.sPRS__patient_id__c, sPRS__sirono_id__c from sPRS__invoice__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Payment Plan
select name, sPRS__sirono_id__c, sPRS__status__c, sPRS__original_balance__c, sPRS__has_error__c, sPRS__error_date__c, sPRS__delinquent__c, sPRS__delinquent_date__c, sPRS__installment_amount__c, sPRS__skip_payment_until__c, sPRS__active__c, sPRS__plan_value_modified_date__c, sPRS__execute_on_day__c, sPRS__plan_type__c, sPRS__original_installment_count__c, sPRS__plan_origin__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__payment_method__r.sPRS__sirono_id__c from sPRS__payment_plan__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Statement
select name, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__hex_code__c, sPRS__large_statement_id__c, sPRS__original_balance__c, sPRS__original_charge_groups__c, s3sPRS___key__c, sPRS__sirono_id__c, sPRS__small_statement_id__c, sPRS__statement_date__c from sPRS__statement__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Charge Group
select name, sPRS__sirono_id__c, sPRS__external_id__c, sPRS__origin_code__c, sPRS__origin_subcode__c, sPRS__account_type__c, sPRS__active__c, sPRS__inactivation_date__c, sPRS__pre_collection_sent__c, sPRS__sirono_assigned_date__c, sPRS__sPRS__reported_selfpay_balance__c, sPRS__sPRS__reported_insurance_balance__c, sPRS__sPRS__reported_prebill_balance__c, sPRS__sPRS__reported_bad_debt__c, sPRS__date_of_service_start_date__c, sPRS__date_of_service_end_date__c, sPRS__account_sub_status__c, sPRS__payment_plan__r.sPRS__sirono_id__c, sPRS__location__r.sPRS__external_id__c, sPRS__invoice__r.sPRS__sirono_id__c, sPRS__provider__c, sPRS__guarantor__r.sPRS__guarantor_id__c, sPRS__patient__r.sPRS__patient_id__c, sPRS__pre_collection_history__c, sPRS__external_guarantor__r.sPRS__external_guarantor_id__c, sPRS__inactivation_flags__c, sPRS__account_status__c from sPRS__charge_group__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Coverage
select name, sPRS__date_effective__c, sPRS__date_termination__c, sPRS__external_id__c, sPRS__external_last_update__c, sPRS__financial_class__c, sPRS__group_name__c, sPRS__group_number__c, sPRS__member_id__c, sPRS__member_name__c, sPRS__member_relationship__c, sPRS__origin_code__c, sPRS__payor__r.sPRS__unique_id__c, sPRS__plan_name__c, sPRS__priority__c, sPRS__sirono_id__c, sPRS__subscriber_dob__c, sPRS__subscriber_id__c, sPRS__subscriber_name__c, sPRS__subscriber_ssn__c, sPRS__subscriber_sex__c, sPRS__verification_status__c from sPRS__coverage__c

# Service (Service2)
select sPRS__aggregate_id__c, sPRS__cpt_code_modifiers__c, sPRS__cpt_code_override__c, sPRS__cpt_code__c, sPRS__charge_code_description__c, sPRS__charge_code__c, sPRS__external_guarantor__r.sPRS__external_guarantor_id__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__provider__c, sPRS__quantity__c, sPRS__service_date__c, sPRS__service_rollup__r.sPRS__external_id__c from service2__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Adjustment
select sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__aggregate_id__c, sPRS__amount__c, sPRS__description__c, sPRS__payor__r.sPRS__unique_id__c, sPRS__post_date__c, sPRS__external_guarantor__r.sPRS__external_guarantor_id__c from sPRS__adjustment__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Patient Payment
select sPRS__payment_plan__r.sPRS__sirono_id__c, sPRS__sirono_id__c, sPRS__payment_method__r.sPRS__sirono_id__c, sPRS__amount__c, sPRS__method__c, sPRS__processor_transaction_id__c, sPRS__guarantor__r.sPRS__Guarantor_Id__c, sPRS__deposit_date__c, sPRS__statement__r.sPRS__sirono_id__c from sPRS__payment__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# PAS Payment
select sPRS__amount__c, sPRS__deposit_date__c, sPRS__id__c, sPRS__payor__r.sPRS__unique_id__c from sPRS__payment__c where sPRS__id__c <> null

# Service Transaction
select name, sPRS__charge_group__r.sPRS__sirono_id__c, sPRS__amount__c, sPRS__external_id__c, sPRS__post_date__c, sPRS__sirono_id__c, sPRS__transaction_date__c, service2__r.sPRS__aggregate_id__c from sPRS__transaction__c where service2__c <> null and service2__r.sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Adjustment Transaction
select name, sPRS__charge_group__r.sPRS__sirono_id__c, sPRS__amount__c, sPRS__external_id__c, sPRS__post_date__c, sPRS__sirono_id__c, sPRS__transaction_date__c, adjustment2__r.sPRS__aggregate_id__c from sPRS__transaction__c where adjustment2__c <> null and adjustment2__r.sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Patient Payment Transaction
select name, sPRS__charge_group__r.sPRS__sirono_id__c, sPRS__amount__c, sPRS__external_id__c, sPRS__payment__r.sPRS__Sirono_Id__c, sPRS__post_date__c, sPRS__sirono_id__c, sPRS__transaction_date__c from sPRS__transaction__c where sPRS__payment__r.sPRS_Sirono_Id__c <> null and sPRS__charge_group__r.sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# PAS Payment Transaction
select name, sPRS__charge_group__r.sPRS__sirono_id__c, sPRS__amount__c, sPRS__external_id__c, sPRS__payment__r.sPRS__id__c, sPRS__post_date__c, sPRS__sirono_id__c, sPRS__transaction_date__c from sPRS__transaction__c where sPRS__payment__r.sPRS_id__c <> null and sPRS__charge_group__r.sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Charge Group Coverage Junction
select name, sPRS__charge_group__r.sPRS__sirono_id__c, sPRS__coverage__r.sPRS__sirono_id__c, sPRS__sirono_id__c from sPRS__charge_group_coverage_junction__c where sPRS__charge_group__r.sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000

# Patient Coverage Junction
select sPRS__coverage__r.sPRS__sirono_id__c, sPRS__patient__r.sPRS__patient_id__c, sPRS__sirono_id__c from sPRS__patient_coverage_junction__c and sPRS__patient__c in (select sPRS__patient__c from sPRS__charge_group__c where sPRS__guarantor__r.sPRS__Guarantor_Id__c >= 200000)

# Contact (Notes)
select sPRS__Guarantor_Id__c, sPRS__notes__c from contact where Guarantor_Id__c >= 200000
